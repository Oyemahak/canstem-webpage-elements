add_filter( 'the_password_form', function( $form ) {
    $extra = '<p><strong>Note:</strong> This password is only for enrolled students and CanSTEM staff. Please do not share it with anyone outside our school community.</p>';
    $extra .= '<p>Need help? <a href="https://canstemeducation.com/contact-us/">Visit our Contact Us page</a>.</p>';
    return $form . $extra;
});
